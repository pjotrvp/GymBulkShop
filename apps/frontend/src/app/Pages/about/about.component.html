Supplement

  • name: string
  • Type: enum
  • image: string
  • ContainsLactose: boolean
  • IsVegan: boolean
  • Price: number
  • flavours: string[]
  • sizes: string[]
  • ingredients: string[]
  • review: Review[]
  • orders: Order[id]
  • createdById: UserId

Bundle

  • name: string
  • description: string
  • image: string
  • Price: number
  • review: Review[]
  • orders: Order[id]
  • createdById: UserId

Review

  • Product: string
  • user: User
  • text: string
  • rating: number

User

  • name: string
  • email: string
  • password: string
  • role: enum
  • orders: Order[id]
  • supplements: Supplement[id]
  • reviews: Review[id]
  • bundles: Bundle[id]

Order

  • user: User
  • supplements: Supplement[id]
  • bundles: Bundle[id]
  • Price: number
  • date: Date
